394     /**
395      * Finds a JsonValueProcessor.<br>
396      * It will search the registered JsonValueProcessors in the following order:
397      * <ol>
398      * <li>beanClass, key</li>
399      * <li>beanClass, type</li>
400      * <li>key</li>
401      * <li>type</li>
402      * </ol>
403      * Returns null if none is registered.<br>
404      * [Java -&gt; JSON]
405      *
406      * @param beanClass    the class to which the property may belong
407      * @param propertyType the type of the property
408      * @param key          the name of the property which may belong to the target class
409      */
410     public JsonValueProcessor findJsonValueProcessor(Class beanClass, Class propertyType, String key) {
411         JsonValueProcessor jsonValueProcessor = null;
412         jsonValueProcessor = (JsonValueProcessor) beanKeyMap.get(beanClass, key);
413         if (jsonValueProcessor != null) {
414             return jsonValueProcessor;
415         }
416 
417         jsonValueProcessor = (JsonValueProcessor) beanTypeMap.get(beanClass, propertyType);
418         if (jsonValueProcessor != null) {
419             return jsonValueProcessor;
420         }
421 
422         jsonValueProcessor = (JsonValueProcessor) keyMap.get(key);
423         if (jsonValueProcessor != null) {
424             return jsonValueProcessor;
425         }
426 
427         Object tkey = jsonValueProcessorMatcher.getMatch(propertyType, typeMap.keySet());
428         jsonValueProcessor = (JsonValueProcessor) typeMap.get(tkey);
429         if (jsonValueProcessor != null) {
430             return jsonValueProcessor;
431         }
432 
433         return null;
434     }
