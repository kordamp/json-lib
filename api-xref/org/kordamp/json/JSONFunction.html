63      public JSONFunction(String[] params, String text) {
64          this.text = (text != null) ? text.trim() : "";
65          if (params != null) {
66              if (params.length == 1 && params[0].trim()
67                  .equals("")) {
68                  this.params = EMPTY_PARAM_ARRAY;
69              } else {
70                  this.params = new String[params.length];
71                  System.arraycopy(params, 0, this.params, 0, params.length);
72                  // remove empty spaces
73                  for (int i = 0; i < params.length; i++) {
74                      this.params[i] = this.params[i].trim();
75                  }
76              }
77          } else {
78              this.params = EMPTY_PARAM_ARRAY;
79          }
80      }

82      /**
83       * Constructs a JSONFunction from a text representation
84       */
85      public static JSONFunction parse(String str) {
86          if (!JSONUtils.isFunction(str)) {
87              throw new JSONException("String is not a function. " + str);
88          } else {
89              String params = JSONUtils.getFunctionParams(str);
90              String text = JSONUtils.getFunctionBody(str);
91              return new JSONFunction((params != null) ? StringUtils.split(params, ",") : null, text != null ? text : "");
92          }
93      }
